@dp.message(F.web_app_data) async def handle_web_app_data(message: Message): """Обработка данных из WebApp""" user_id = message.from_user.id web_app_data = message.web_app_data # Подробное логирование logger.info("=" * 50) logger.info(f"📱 WebApp data received from user {user_id}") logger.info(f"Button text: {web_app_data.button_text}") logger.info(f"Data: {web_app_data.data}") logger.info(f"Raw data type: {type(web_app_data.data)}") logger.info("=" * 50) # Отправляем подтверждение получения await message.answer( f"✅ **Данные получены!**\n\n" f"`{web_app_data.data}`", parse_mode="Markdown" ) try: # Парсим данные data = json.loads(web_app_data.data) logger.info(f"Parsed JSON: {data}") if data.get("type") == "qr_scanned": qr_code = data.get("code") await message.answer(f"✅ QR код: `{qr_code}`") except json.JSONDecodeError as e: logger.error(f"JSON decode error: {e}") await message.answer(f"❌ Не JSON: `{web_app_data.data}`") except Exception as e: logger.error(f"Error: {e}") await message.answer(f"❌ Ошибка: {e}")
